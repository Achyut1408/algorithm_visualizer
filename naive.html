def naive_string_matching(text, pattern):
    n = len(text)
    m = len(pattern)
    positions = []

    for i in range(n - m + 1):
        j = 0
        while j < m:
            if text[i + j] != pattern[j]:
                break
            j += 1
        if j == m:
            positions.append(i)

    return positions

import java.util.ArrayList;

public class NaiveStringMatching {
    public static ArrayList naiveStringMatching(String text, String pattern) {
        int n = text.length();
        int m = pattern.length();
        ArrayList positions = new ArrayList<>();

        for (int i = 0; i <= n - m; i++) {
            int j;
            for (j = 0; j < m; j++) {
                if (text.charAt(i + j) != pattern.charAt(j))
                    break;
            }
            if (j == m)
                positions.add(i);
        }

        return positions;
    }

    public static void main(String[] args) {
        String text = System.console().readLine("Enter the text: ");
        String pattern = System.console().readLine("Enter the pattern to search: ");

        ArrayList positions = naiveStringMatching(text, pattern);

        if (!positions.isEmpty()) {
            System.out.println("Pattern found at positions: " + positions);
        } else {
            System.out.println("Pattern not found in the text.");
        }
    }
}
